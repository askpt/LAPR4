021package csheets.core.formula.lang;
022
023import csheets.core.formula.Expression;
024
025/**
026 * An exception that is thrown when a division by zero is evaluated.
027 * @author Einar Pehrson
028 */
029public class DivisionByZeroException extends Exception {
030
031        /** The serialVersionUID of the DivisionByZeroException.java */
032        private static final long serialVersionUID = -5387697381170031793L;
033
034        /**
035         * Creates a new division-by-zero exception.
036         * @param expression the expression that evaluated to zero
037         */
038        public DivisionByZeroException(Expression expression) {
039                super("Illegal divisor (" + expression + "): division by 0 is undefined");
040        }
041
042        /**
043         * Returns a string representation of the exception.
044         * @return a string representation of the exception
045         */
046        public String toString() {
047                return "#DIV/0!";
048        }
049}
