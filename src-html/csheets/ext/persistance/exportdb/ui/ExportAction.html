017
018    /** The user interface controller */

021    /**
022     * Creates a new action.
023     * 
024     * @param uiController
025     *            the user interface controller
026     */
027    public ExportAction(UIController uiController)

029        this.uiController = uiController;
030    }
031
032    @Override
033    protected String getName()

035        return "Export";
036    }
037
038    @Override
039    protected void defineProperties()

041    }
042
043    /**
044     * A simple action that presents a confirmation dialog. this method catch

046     * 
047     * @param event
048     *            the event that was fired
049     */
050    @Override
051    public void actionPerformed(ActionEvent event) 
052    {
053        Cell[][] cells = focusOwner.getSelectedCells();
054        try 
055        {
056                UIExport uiExp = new UIExport(cells);
057        } 
058        catch (Exception e2)

060                e2.printStackTrace();
061        }
