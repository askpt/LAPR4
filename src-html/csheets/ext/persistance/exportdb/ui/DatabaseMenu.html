009/**

013@Deprecated
014public class DatabaseMenu extends JMenu 
015{
016
017        /**
018         * 
019         */
020        private static final long serialVersionUID = 1069502879234788510L;
021
022        /**
023         * Creates a new simple menu.
024         * This constructor creates and adds the menu options. 
025         * In this simple example only one menu option is created.
026         * A menu option is an action (in this case {@link csheets.ext.simple.ui.ExampleAction})
027         * @param uiController the user interface controller
028         */
029        public DatabaseMenu(UIController uiController) 
030        {
031                super("Connect with database (deprecated)");
032                setMnemonic(KeyEvent.VK_E);
033
034                // Adds actions
035                add(new ExportAction(uiController));
036                add(new ImportAction(uiController));
037                add(new UpdateAction(uiController));
038        }
