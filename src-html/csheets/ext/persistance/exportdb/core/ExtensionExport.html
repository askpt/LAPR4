001package csheets.ext.persistance.exportdb.core;

003import csheets.ext.Extension;
004import csheets.ext.persistance.exportdb.ui.UIDatabaseExtension;
005import csheets.ext.simple.ui.UIExtensionExample;
006import csheets.ui.ctrl.UIController;
007import csheets.ui.ext.UIExtension;
008
009/**
010 * A export extension just to show how the extension mechanism works.
011 * An extension must extend the Extension abstract class.
012 * The class that implements the Extension is the "bootstrap" of the extension.
013 * @see Extension
014 * @author 1110333 Tiago Pacheco
015 */
016@Deprecated
017public class ExtensionExport extends Extension 
018{
019
020        /** The name of the extension */
021        public static final String NAME = "Export";
022
023        /**
024         * Creates a new Example extension.
025         */
026        public ExtensionExport() 
027        {
028                super(NAME);
029        }
030        
031        /**
032         * Returns the user interface extension of this extension (an instance of the class {@link  csheets.ext.simple.ui.UIExtensionExample}). <br/>
033         * In this extension example we are only extending the user interface.
034         * @param uiController the user interface controller
035         * @return a user interface extension, or null if none is provided
036         */
037        public UIExtension getUIExtension(UIController uiController) 
038        {
039                return new UIDatabaseExtension(this, uiController);
040        }
041}
