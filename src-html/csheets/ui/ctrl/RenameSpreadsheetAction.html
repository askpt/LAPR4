021package csheets.ui.ctrl;
022
023import java.awt.event.ActionEvent;
024
025import javax.swing.ImageIcon;
026
027import csheets.CleanSheets;
028
029/**
030 * A spreadsheet insertion operation.
031 * @author Einar Pehrson
032 */
033@SuppressWarnings("serial")
034public class RenameSpreadsheetAction extends FocusOwnerAction {
035
036        /** The user interface controller */
037        @SuppressWarnings("unused")
038        private UIController uiController;
039
040        /**
041         * Creates a new spreadsheet insertion action.
042         * @param uiController the user interface controller
043         */
044        public RenameSpreadsheetAction(UIController uiController) {
045                this.uiController = uiController;
046        }
047
048        protected String getName() {
049                return "Rename";
050        }
051
052        protected void defineProperties() {
053                setEnabled(false);
054                putValue(SMALL_ICON, new ImageIcon(CleanSheets.class.getResource("res/img/rename_sheet.gif")));
055        }
056
057        /**
058         * Renames the spreadsheet of the focus owner.
059         * @param event the event that was fired
060         */
061        public void actionPerformed(ActionEvent event) {
062                if (focusOwner == null)
063                        return;
064        }
065}
